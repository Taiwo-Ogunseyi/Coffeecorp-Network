Configuration command

# Switch VLAN Configuration
1. enable
2. configure terminal
3. vlan 10
4. exit

# Repeat steps 1 - 4 to create other VLANs

5. do show VLAN (optional)
6. interface range fa0/1-2
7. switchport mode access
8. switchport access vlan 10

# Repeat steps 5 - 8 for the other vlans created

9. exit
10. end



# Router VLAN Configuration
1. enable
2. configure terminal
3. interface g0/0.10
4. encapsulation dot1Q 10
5. ip address 192.168.10.1 255.255.255.0
6. exit

# Repeat steps 3 - 6 for the other vlans, using their respective ip address




# Switch VLAN Configuration
1. enable
2. configure terminal
3. interface g0/1 (the interface on switch)
4. switchport mode trunk



# Router ACL Configuration
1. enable 
2. configure terminal
3. access-list 100 deny ip 192.168.10.0 0.0.0.255 host 192.168.30.2
4. access-list 100 permit any any
5. interface g0/0.10 (the interface we want to deny)
6. ip access-group 100 in
7. exit
8. exit
9. show access-list


#Router Telnet/Remote Access Configuration
1. enable 
2. configure terminal
3. access-list 110 permit ip 192.168.20.0 0.0.0.255 any
4. access-list 110 deny ip any any
5. line vty 0 4
6. access-class 110 in
7. password cisco (you can choose any password)
8. login
9. exit
10. enable password cisco
11. exit
12. write memory
13. end
